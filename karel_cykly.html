{

        {        
                World.readWorld("test.txt");
                World.setVisible(true);  // zobrazime dvorecek
                UrRobot kaja = new UrRobot(1, 1, East, 3);  // vytvorime robota se jmenem kaja
                  // na souradnicich [1, 1] smerujiciho na vychod a nesouciho tri kameny (beeper)
                
                int i = 1;  // vytvorime celociselnou promennou (pocitadlo) a vlozime do ni cislo 1
                while (i < 5)  // dokud je hodnota promenn "i" mensi ne 5 opakuj  
                {
                      kaja.move(); 
                      // System.out.println("Kaja udìlal krok èíslo " + i);
                      i++;  //  hodnotu pocitadla zvetsime o jednicku 
                }
                kaja.turnOff();  // kaja se vypne (da si slofika :-)
        }                
}
