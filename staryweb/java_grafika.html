import java.awt.event.*;

class Platno extends Canvas {
  public void paint(Graphics g) {
    g.drawLine(15, 20, 100, 200);
    g.drawOval(300, 200, 50, 80);
    g.setColor(Color.red);
    g.drawRect(450, 300, 50, 80);
    g.setColor(Color.blue);
    g.fillOval(500, 200, 50, 120);
    g.fillRect(650, 300, 50, 120);
    g.drawString("Èeština se vypisuje správnì :-)", 100, 400);

  }
}

public class Kresleni extends Frame {
  Kresleni() {
    super.setTitle(getClass().getName());
    Platno pl = new Platno();
    this.add(pl, BorderLayout.CENTER);
    this.setSize(800, 600);

    addWindowListener(new WindowAdapter() {
      public void windowClosing(WindowEvent e) {
        System.exit(1);
      }
    });
  }

  public static void main(String[] args) {
    new Kresleni().setVisible(true);
  }
}
